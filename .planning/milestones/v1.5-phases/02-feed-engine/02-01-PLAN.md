   ```typescript
   import { ObjectType, Field, Float } from '@nestjs/graphql';

   @ObjectType()
   export class CityCoordinates {
     @Field(() => Float) lat: number;
     @Field(() => Float) lng: number;
     @Field() city: string;
     @Field(() => String, { nullable: true }) country?: string;
   }

   @ObjectType()
   export class CitySuggestion {
     @Field() name: string;        // "Austin, Texas, United States"
     @Field(() => Float) lat: number;
     @Field(() => Float) lng: number;
   }
   ```

3. **Create `backend/src/geocoding/geocoding.service.ts`:**
   - Inject `PrismaService` and `ConfigService`
   - Initialize Mapbox geocoding client from `MAPBOX_ACCESS_TOKEN` env var
   - If no token, log warning and degrade gracefully (return null/empty)

   Methods:
   - `geocodeCity(cityName: string): Promise<CityCoordinates | null>` - Forward geocode with DB cache.
     Check CityLocation table first (`prisma.cityLocation.findFirst({ where: { cityName } })`).
     If found, return cached. If not, call Mapbox `forwardGeocode({ query: cityName, limit: 1, types: ['place'] })`,
     store result in CityLocation, return coordinates. Handle errors gracefully (log, return null).

   - `reverseGeocode(lat: number, lng: number): Promise<string>` - GPS to city name for location badge (FEED-07).
     Call Mapbox `reverseGeocode({ query: [lng, lat], limit: 1, types: ['place'] })`.
     Note: Mapbox uses [lng, lat] order! Return city name or 'Unknown Location' on error.

   - `searchCities(query: string, limit: number = 5): Promise<CitySuggestion[]>` - Autocomplete for manual location change (FEED-08).
     Call Mapbox `forwardGeocode({ query, limit, types: ['place'], autocomplete: true })`.
     Map features to CitySuggestion objects. Return empty array on error.
