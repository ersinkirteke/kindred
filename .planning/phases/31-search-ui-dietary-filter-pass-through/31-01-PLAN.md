       ```swift
       // Diet chips map to Spoonacular `diets` param
       private let chipToSpoonacularDiet: [String: String] = [
           "Vegan": "vegan",
           "Vegetarian": "vegetarian",
           "Keto": "ketogenic",
           "Pescatarian": "pescetarian",
           "Low-Carb": "paleo",
           "Halal": "halal",
           "Kosher": "kosher"
       ]
       // Intolerance chips map to Spoonacular `intolerances` param
       private let chipToSpoonacularIntolerance: [String: String] = [
           "Gluten-Free": "gluten",
           "Dairy-Free": "dairy",
           "Nut-Free": "tree nut"
       ]
       public func mapChipsToSearchParams(_ chips: Set<String>) -> (diets: [String], intolerances: [String]) {
           var diets: [String] = []
           var intolerances: [String] = []
           for chip in chips {
               if let diet = chipToSpoonacularDiet[chip] { diets.append(diet) }
               else if let intolerance = chipToSpoonacularIntolerance[chip] { intolerances.append(intolerance) }
           }
           return (diets, intolerances)
       }
       ```

    8. Implement search action handlers in the reducer body:

       **`.searchQueryChanged(query)`:**
       - Set `state.searchQuery = query`
       - If `query.isEmpty`: set `feedMode = .browse`, clear searchResults, cancel debounce, return
       - Set `feedMode = .search`
       - If `query.count < 3`: clear searchResults, return `.cancel(id: SearchDebounceID.self)`
       - Set `isSearching = true`, clear `searchResults` immediately (no stale results)
       - Return `.run` with `Task.sleep(nanoseconds: 300_000_000)` then `send(.executeSearch(query: query, filters: state.activeDietaryFilters, cursor: nil))`, wrapped in `.cancellable(id: SearchDebounceID.self, cancelInFlight: true)`

       **`.executeSearch(query, filters, cursor)`:**
       - Map `filters` to Spoonacular params using `mapChipsToSearchParams(filters)`
       - Build `SearchRecipesInput`: query, diets (if non-empty), intolerances (if non-empty), first: 20, after: cursor
       - Fetch via `apolloClient.fetch(SearchRecipesQuery(input: input))`
       - Map edges to `[RecipeCard]` using `RecipeCard.from(searchRecipe:)`
       - Extract `pageInfo.endCursor`, `pageInfo.hasNextPage`, `totalCount`
       - Send `.searchResultsLoaded(.success((cards, endCursor, hasNextPage, totalCount)))`
       - On error: send `.searchResultsLoaded(.failure(error))`

       **`.searchResultsLoaded(.success(cards, cursor, hasMore, total))`:**
       - Set `isSearching = false`
       - If `cursor != nil` (pagination): append cards to existing `searchResults`
       - Else (fresh search): replace `searchResults` with cards
       - Set `searchEndCursor = cursor`, `searchHasNextPage = hasMore`, `searchTotalCount = total`

       **`.searchResultsLoaded(.failure)`:**
       - Set `isSearching = false`
       - Check if error message contains "quota" — if yes, set `isQuotaExhausted = true`
       - Otherwise set `state.error` with a user-friendly message

       **`.loadMoreSearchResults`:**
       - Guard `searchHasNextPage` and `!isSearching`
       - Set `isSearching = true`
       - Send `.executeSearch(query: state.searchQuery, filters: state.activeDietaryFilters, cursor: state.searchEndCursor)`

       **`.clearSearch`:**
       - Reset: `feedMode = .browse`, `searchQuery = ""`, `searchResults = []`, `isSearching = false`, `searchEndCursor = nil`, `searchHasNextPage = true`, `searchTotalCount = 0`, `isQuotaExhausted = false`
       - Cancel debounce: `.cancel(id: SearchDebounceID.self)`
