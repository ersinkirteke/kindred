    **ApolloClientFactory.swift** — Create and configure ApolloClient:
    ```swift
    import Apollo
    import ApolloSQLite
    import Foundation

    public final class ApolloClientFactory {
        public static func create(tokenProvider: @escaping () async -> String?) -> ApolloClient {
            // 1. SQLite cache for offline-first
            let documentsURL = FileManager.default.urls(for: .documentDirectory, in: .userDomainMask).first!
            let sqliteFileURL = documentsURL.appendingPathComponent("kindred_apollo_cache.sqlite")

            let sqliteCache = try! SQLiteNormalizedCache(fileURL: sqliteFileURL)
            let store = ApolloStore(cache: sqliteCache)

            // 2. Auth interceptor
            let authInterceptor = AuthInterceptor(tokenProvider: tokenProvider)

            // 3. Network transport
            let url = URL(string: "https://api.kindred.app/graphql")! // TODO: Configure per environment
            let interceptorProvider = DefaultInterceptorProvider(store: store)
            let transport = RequestChainNetworkTransport(
                interceptorProvider: interceptorProvider,
                endpointURL: url,
                additionalHeaders: [:]
            )

            return ApolloClient(networkTransport: transport, store: store)
        }
    }
    ```

    NOTE: The exact Apollo 2.0 API for custom interceptors may differ from 0.x/1.x. Consult the Apollo iOS 2.0 docs for the correct interceptor chain pattern. The key pattern is: inject JWT from Clerk into Authorization header before each request.

    **AuthInterceptor.swift** — Inject Clerk JWT into requests:
    - Receives a `tokenProvider: () async -> String?` closure (injected from AuthClient)
    - Before each request: call tokenProvider to get fresh JWT
    - Add `Authorization: Bearer {token}` header
    - If no token (guest user): proceed without auth header (backend allows unauthenticated queries for feed)
    - Implement Apollo's interceptor protocol for v2.0

    **CacheConfig.swift** — Cache policy configuration:
    ```swift
    public enum CachePolicy {
        /// Fetch from cache first, then update from network
        public static let offlineFirst: Apollo.CachePolicy = .returnCacheDataAndFetch
        /// Network only, no cache
        public static let networkOnly: Apollo.CachePolicy = .fetchIgnoringCacheData
        /// Cache only (for truly offline scenarios)
        public static let cacheOnly: Apollo.CachePolicy = .returnCacheDataDontFetch
    }
    ```
    Default policy: `returnCacheDataAndFetch` for offline-first UX (per research recommendation).

    **ClerkAuthClient.swift** — Clerk SDK wrapper:
    ```swift
    import ClerkKit

    public final class ClerkAuthClient {
        public init() {}

        /// Get current JWT token (nil if not authenticated)
        public func getToken() async -> String? {
            try? await Clerk.shared.session?.getToken()?.jwt
        }

        /// Check if user is authenticated
        public var isAuthenticated: Bool {
            Clerk.shared.session != nil
        }

        /// Current user info (nil if guest)
        public var currentUser: ClerkUser? {
            guard let user = Clerk.shared.user else { return nil }
            return ClerkUser(
                id: user.id,
                email: user.emailAddresses.first?.emailAddress ?? "",
                displayName: user.firstName ?? user.username ?? ""
            )
        }
    }
    ```
