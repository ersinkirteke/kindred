    ```typescript
    async getExpiringItems(params: { daysAhead: number }): Promise<PantryItemModel[]> {
      const now = new Date();
      const futureDate = new Date();
      futureDate.setDate(now.getDate() + params.daysAhead);

      const items = await this.prisma.pantryItem.findMany({
        where: {
          expiryDate: { gte: now, lte: futureDate },
          isDeleted: false,
        },
        orderBy: { expiryDate: 'asc' },
      });

      return items.map(this.toPantryItemModel);
    }
    ```

    Add `estimateAndSetExpiry` method that uses ExpiryEstimatorService (injected) to auto-set expiryDate on newly added items when user doesn't provide one:
    ```typescript
    async estimateAndSetExpiry(itemId: string): Promise<void> {
      const item = await this.prisma.pantryItem.findUnique({ where: { id: itemId } });
      if (!item || item.expiryDate) return; // Already has expiry or not found

      const shelfLifeDays = await this.expiryEstimator.estimateExpiryDate(
        item.normalizedName || item.name,
        item.storageLocation
      );
      const expiryDate = new Date();
      expiryDate.setDate(expiryDate.getDate() + shelfLifeDays);

      await this.prisma.pantryItem.update({
        where: { id: itemId },
        data: { expiryDate },
      });
    }
    ```
